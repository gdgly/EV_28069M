238 static inline float_t FILTER_SO_run(FILTER_SO_Handle handle,const float_t inputValue)
239 {
240  FILTER_SO_Obj *obj = (FILTER_SO_Obj *)handle;
241 
242  float_t a1 = obj->a1;
243  float_t a2 = obj->a2;
244  float_t b0 = obj->b0;
245  float_t b1 = obj->b1;
246  float_t b2 = obj->b2;
247  float_t x1 = obj->x1;
248  float_t x2 = obj->x2;
249  float_t y1 = obj->y1;
250  float_t y2 = obj->y2;
251 
252 
253  // compute the output
254  float_t y0 = (b0 * inputValue) + (b1 * x1) + (b2 * x2) - (a1 * y1) - (a2 * y2);
255 
256 
257  // store values for next time
258  obj->x1 = inputValue;
259  obj->x2 = x1;
260  obj->y1 = y0;
261  obj->y2 = y1;
262 
263  return(y0);
264 } // end of FILTER_SO_run() function
