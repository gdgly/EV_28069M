139 static inline void IPARK_run(IPARK_Handle handle,const MATH_vec2 *pInVec,MATH_vec2 *pOutVec)
140 {
141  IPARK_Obj *obj = (IPARK_Obj *)handle;
142 
143  float_t sinTh = obj->sinTh;
144  float_t cosTh = obj->cosTh;
145 
146  float_t value_0 = pInVec->value[0];
147  float_t value_1 = pInVec->value[1];
148 
149  pOutVec->value[0] = (value_0 * cosTh) - (value_1 * sinTh);
150  pOutVec->value[1] = (value_1 * cosTh) + (value_0 * sinTh);
151 
152  return;
153 } // end of IPARK_run() function
